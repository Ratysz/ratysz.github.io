    pub dimensions: (u32, u32),

A builder structure containing window settings that can be set at runtime and changed with graphics::set_mode().

Defaults:

extern crate ggez;
use ggez::conf::{FullscreenType, WindowMode};

assert_eq!(
    WindowMode::default(),
    WindowMode {
        dimensions: (800, 600),
        min_dimensions: None,
        max_dimensions: None,
        fullscreen_type: FullscreenType::Off,
        maximized: false,
        hidden: false,
        borderless: false,
        always_on_top: false,
    },
);

Fields

dimensions: (u32, u32)

Window's inner dimensions (drawable area).

min_dimensions: Option<(u32, u32)>

Window's minimum dimensions; None is no limit.

max_dimensions: Option<(u32, u32)>

Window's maximum dimensions; None is no limit.

fullscreen_type: FullscreenType

Fullscreen type.

maximized: bool

Whether or not to maximize the window.

hidden: bool

Whether or not the window is hidden.

borderless: bool

Whether or not to show window decorations.

always_on_top: bool

Whether or not the window is pinned to always be on top of other windows.
